  📈 实时查看输出的技巧

  技巧1：使用tee同时显示和保存

  python test_double_helix.py 2>&1 | tee output.txt

  技巧2：使用tail -f实时追踪日志

  # 先运行测试到后台
  python monitor_v2_emergence.py > monitor.log 2>&1 &

  # 然后实时查看
  tail -f monitor.log

  技巧3：使用less分页查看

  python smoke_test.py 2>&1 | tee output.txt
  # 测试完成后
  less output.txt
  # 按`q`退出

  技巧4：过滤关键信息

  # 只看错误信息
  python smoke_test.py 2>&1 | grep -i "error\|fail"

  # 只看关键指标
  python smoke_test.py 2>&1 | grep -i "质量\|performance\|通过"

  ---
  🔍 进程管理命令

  # 查看Python进程
  ps aux | grep python

  # 查看特定进程
  ps aux | grep "monitor_v2_emergence"

  # 终止进程（优雅退出）
  kill <PID>

  # 强制终止
  kill -9 <PID>

  # 后台运行并保存PID
  python monitor_v2_emergence.py > monitor.log 2>&1 & echo $! > monitor.pid

  # 使用PID终止
  kill $(cat monitor.pid)
