```python
async def _validate_tool_call(self, tool_name: str, params: Dict) -> Dict:
    """工具调用前验证"""

    # 1. 检查工具是否注册
    if tool_name not in self.tools:
        return {'valid': False, 'error': f'未注册的工具: {tool_name}'}

    # 2. 检查必需参数
    tool_info = self.tool_capabilities.get(tool_name, {})
    required = tool_info.get('required_params', {}).get('execute', [])
    missing = [p for p in required if p not in params]
    if missing:
        return {'valid': False, 'error': f'缺少必需参数: {missing}'}

    # 3. 特定工具的验证
    if tool_name == 'local_document_reader':
        method = params.get('_method')
        if method == 'read':
            path = params.get('path')
            if path and not Path(path).exists():
                return {'valid': False, 'error': f'文件不存在: {path}'}

    return {'valid': True}
```

#### 2. 构建文档索引（优先级：中）

```python
class DocumentIndex:
    """文档索引 - 加速文档搜索"""

    def __init__(self, root_path: str):
        self.root_path = Path(root_path)
        self.index_file = root_path / '.cache' / 'doc_index.json'
        self.index = self._load_index()

    def build_index(self):
        """增量构建索引"""
        index = {}
        for doc_path in self.root_path.rglob('*.md'):
            # 提取关键词
            keywords = self._extract_keywords(doc_path)
            for kw in keywords:
                if kw not in index:
                    index[kw] = []
                index[kw].append(str(doc_path))

        # 保存索引
        self.index = index
        self._save_index()

    def search(self, query: str) -> List[str]:
        """快速搜索"""
        return self.index.get(query, [])
```
